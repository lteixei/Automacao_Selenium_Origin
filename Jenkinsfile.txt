    agent any

    tools { nodejs 'Node23' }

    stages {

            steps {
                sh 'npm i'
            }
        }
        stage('Running production backtests') {
            steps {
                sh 'mvn clean test'
            }
        }
    }
    post {
        always {
            junit 'results/*.xml'
        }
    }
}
